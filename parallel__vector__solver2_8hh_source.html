59 template <typename T>
60 inline int solve_parallel_vector2_simplex1(const T VV[2][2], const T WW[2][2],
61  T lambda[2], T mu[2][2], const T epsilon = std::numeric_limits<T>::epsilon())
62 {
63  T V[2][2], W[2][2];
64  transpose2x2(VV, V);
65  transpose2x2(WW, W);
66 
67  T P[3];
68  characteristic_polynomial_2x2(V, W, P);
69 
70  T l[2] = {0};
71  const int n_roots = solve_quadratic_real(P, l, epsilon);
72  int n_solutions = 0;
73 
74  for (int i = 0; i < n_roots; i ++) {
75  if (std::abs(l[i]) <= epsilon) continue;
76 
77  const T a[2] = {
78  (V[0][0] - V[0][1]) - l[i] * (W[0][0] - W[0][1]),
79  (V[1][0] - V[1][1]) - l[i] * (W[1][0] - W[1][1])
80  };
81  const T b[2] = {
82  -(V[0][1] - l[i] * W[0][1]),
83  -(V[1][1] - l[i] * W[1][1])
84  };
85 
86  T nu[2];
87  const auto cond = solve_least_square2x1(a, b, nu[0]);
88  nu[1] = T(1) - nu[0];
89 
90  if (nu[0] >= -epsilon && nu[0] <= 1+epsilon &&
91  nu[1] >= -epsilon && nu[1] <= 1+epsilon)
92  {
93  double v[2], w[2];
94  ftk::linear_interpolation_1simplex_vector2(VV, nu, v);
95  ftk::linear_interpolation_1simplex_vector2(WW, nu, w);
96  const double c = ftk::cross_product2(v, w);
97  if (c > 1e-2) {
98  fprintf(stderr, "rejecting: nu={%f, %f}, v={%f, %f}, w={%f, %f}, c=%f\n",
99  nu[0], nu[1], v[0], v[1], w[0], w[1], c);
100  print2x2("V", VV);
101  print2x2("W", WW);
102  continue;
103  }
104 
105  const int j = n_solutions ++;
106  lambda[j] = l[i];
107  mu[j][0] = nu[0];
108  mu[j][1] = nu[1];
109  }
110  }
111 
112  return n_solutions;
113 }

123  T det = ftk::matrix_inverse2x2(A, invA);

133 template <typename T>
134 inline void characteristic_polynomials_parallel_vector2_simplex2(const T V[3][2], const T W[3][2], T Q[3], T P[3][3])
135 {
136  const T A[2][2] = { // linear transformation
137  {V[0][0] - V[2][0], V[1][0] - V[2][0]},
138  {V[0][1] - V[2][1], V[1][1] - V[2][1]}
139  };
140  const T B[2][2] = {
141  {W[0][0] - W[2][0], W[1][0] - W[2][0]},
142  {W[0][1] - W[2][1], W[1][1] - W[2][1]}
143  };
144  const T a[2] = {V[2][0], V[2][1]};
145  const T b[2] = {W[2][0], W[2][1]};
146 
147  const T rhs[2][2] = {
148  {-a[0], b[0]},
149  {-a[1], b[1]}
150  };
151 
152  // coefs for Q
153  characteristic_polynomial_2x2(A, B, Q);
154 
155  T adj[2][2][2] = { // adjugate matrix
156  {{ A[1][1], -B[1][1]}, {-A[0][1], B[0][1]}},
157  {{-A[1][0], B[1][0]}, { A[0][0], -B[0][0]}}
158  };
159 
160  for (int i = 0; i < 2; i ++) {
161  T poly[3] = {0};
162  for (int j = 0; j < 3; j ++)
163  P[i][j] = 0; // clear results
164  for (int j = 0; j < 2; j ++) {
165  polynomial_multiplication(adj[i][j], 1, rhs[j], 1, poly);
166  polynomial_addition_in_place(P[i], 2, poly, 2);
167  }
168  }
169 
170  P[2][0] = Q[0] - P[0][0] - P[1][0];
171  P[2][1] = Q[1] - P[0][1] - P[1][1];
172  P[2][2] = Q[2] - P[0][2] - P[1][2];
173 }
