167  const T V[2][3], const T W[2][3],
168  T lambda[2], T mu[2][2],
169  const T epsilon = std::numeric_limits<T>::epsilon()) // 1e-9)
170 {
171  const T P[3][2] = { // rhs
172  {V[1][0], -W[1][0]},
173  {V[1][1], -W[1][1]},
174  {V[1][2], -W[1][2]}
175  };
176  const T norm = matrix_frobenius_norm_real3x2(P);
177  if (norm <= epsilon)
178  return std::numeric_limits<int>::max(); // unlimited.
179 
180  const T Q[3][2] = { // lhs
181  {V[1][0]-V[0][0], W[0][0]-W[1][0]},
182  {V[1][1]-V[0][1], W[0][1]-W[1][1]},
183  {V[1][2]-V[0][2], W[0][2]-W[1][2]}
184  };

193  polynomial_multiplication(P[2], 1, Q[1], 1, F[0]);
194  polynomial_multiplication(P[0], 1, Q[2], 1, E[1]);
195  polynomial_multiplication(P[2], 1, Q[0], 1, F[1]);
196  polynomial_multiplication(P[0], 1, Q[1], 1, E[2]);
197  polynomial_multiplication(P[1], 1, Q[0], 1, F[2]);

201  polynomial_subtraction(E[i], 2, F[i], 2, H[i]);

204  cross_product(H[0], H[1], C);

243 template <typename T>
244 inline void characteristic_polynomials_parallel_vector3_simplex3(const T V[4][3], const T W[4][3], T Q[4], T P[4][4])
245 {
246  const T A[3][3] = { // linear transformation
247  {V[0][0] - V[3][0], V[1][0] - V[3][0], V[2][0] - V[3][0]},
248  {V[0][1] - V[3][1], V[1][1] - V[3][1], V[2][1] - V[3][1]},
249  {V[0][2] - V[3][2], V[1][2] - V[3][2], V[2][2] - V[3][2]}
250  };
251  const T B[3][3] = {
252  {W[0][0] - W[3][0], W[1][0] - W[3][0], W[2][0] - W[3][0]},
253  {W[0][1] - W[3][1], W[1][1] - W[3][1], W[2][1] - W[3][1]},
254  {W[0][2] - W[3][2], W[1][2] - W[3][2], W[2][2] - W[3][2]}
255  };
256  const T a[3] = {V[3][0], V[3][1], V[3][2]};
257  const T b[3] = {W[3][0], W[3][1], W[3][2]};
258 
259  const T rhs[3][2] = { // polynomials for the right hand side vectors
260  {-a[0], b[0]},
261  {-a[1], b[1]},
262  {-a[2], b[2]}};
263 
264  // coefficients for Q
265  characteristic_polynomial_3x3(A, B, Q);
266 
267  T adj[3][3][3]; // build adjugate matrix
268  characteristic_polynomial_2x2(A[1][1], A[1][2], A[2][1], A[2][2], B[1][1], B[1][2], B[2][1], B[2][2], adj[0][0]);
269  characteristic_polynomial_2x2(A[1][0], A[1][2], A[2][0], A[2][2], B[1][0], B[1][2], B[2][0], B[2][2], adj[1][0]);
270  characteristic_polynomial_2x2(A[1][0], A[1][1], A[2][0], A[2][1], B[1][0], B[1][1], B[2][0], B[2][1], adj[2][0]);
271  characteristic_polynomial_2x2(A[0][1], A[0][2], A[2][1], A[2][2], B[0][1], B[0][2], B[2][1], B[2][2], adj[0][1]);
272  characteristic_polynomial_2x2(A[0][0], A[0][2], A[2][0], A[2][2], B[0][0], B[0][2], B[2][0], B[2][2], adj[1][1]);
273  characteristic_polynomial_2x2(A[0][0], A[0][1], A[2][0], A[2][1], B[0][0], B[0][1], B[2][0], B[2][1], adj[2][1]);
274  characteristic_polynomial_2x2(A[0][1], A[0][2], A[1][1], A[1][2], B[0][1], B[0][2], B[1][1], B[1][2], adj[0][2]);
275  characteristic_polynomial_2x2(A[0][0], A[0][2], A[1][0], A[1][2], B[0][0], B[0][2], B[1][0], B[1][2], adj[1][2]);
276  characteristic_polynomial_2x2(A[0][0], A[0][1], A[1][0], A[1][1], B[0][0], B[0][1], B[1][0], B[1][1], adj[2][2]);
277  polynomial_scalar_multiplication(adj[0][1], 2, T(-1));
278  polynomial_scalar_multiplication(adj[1][0], 2, T(-1));
279  polynomial_scalar_multiplication(adj[1][2], 2, T(-1));
280  polynomial_scalar_multiplication(adj[2][1], 2, T(-1));
281 
282  for (int i = 0; i < 3; i ++) {
283  T poly[4] = {0};
284  for (int j = 0; j < 4; j ++)
285  P[i][j] = 0; // clear results
286  for (int j = 0; j <3; j ++) {
287  polynomial_multiplication(adj[i][j], 2, rhs[j], 1, poly);
288  polynomial_addition_in_place(P[i], 3, poly, 3);
289  }
290  }
291 
292  P[3][0] = Q[0] - P[0][0] - P[1][0] - P[2][0];
293  P[3][1] = Q[1] - P[0][1] - P[1][1] - P[2][1];
294  P[3][2] = Q[2] - P[0][2] - P[1][2] - P[2][2];
295  P[3][3] = Q[3] - P[0][3] - P[1][3] - P[2][3];
296 }
